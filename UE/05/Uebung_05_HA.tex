	static void hilbertLinks(Canvas c, int ordnung, int length) {
		if (ordnung <= 0)
			return;
		c.rotate(-90);
		hilbertRechts(c, ordnung - 1, length);
		c.drawForward(length);
		c.rotate(90);
		hilbertLinks(c, ordnung - 1, length);
		c.drawForward(length);
		hilbertLinks(c, ordnung - 1, length);
		c.rotate(90);
		c.drawForward(length);
		hilbertRechts(c, ordnung - 1, length);
		c.rotate(-90);
	}

	static void hilbertRechts(Canvas c, int ordnung, int length) {
		if (ordnung <= 0)
			return;
		c.rotate(90);
		hilbertLinks(c, ordnung - 1, length);
		c.drawForward(length);
		c.rotate(-90);
		hilbertRechts(c, ordnung - 1, length);
		c.drawForward(length);
		hilbertRechts(c, ordnung - 1, length);
		c.rotate(-90);
		c.drawForward(length);
		hilbertLinks(c, ordnung - 1, length);
		c.rotate(90);
	}
